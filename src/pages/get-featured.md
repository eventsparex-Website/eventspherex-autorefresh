<script>
(function(){
  var uploadArea = document.getElementById('upload-area');
  var fileInput = document.getElementById('file-input');
  var fileNameDiv = document.getElementById('file-name');
  var fileNameText = document.getElementById('file-name-text');
  var fileRemove = document.getElementById('file-remove');

  if(!uploadArea) return;

  uploadArea.addEventListener('click', function(){ fileInput.click(); });
  uploadArea.addEventListener('dragover', function(e){ e.preventDefault(); this.style.borderColor='#b3d237'; });
  uploadArea.addEventListener('dragleave', function(){ this.style.borderColor='rgba(255,255,255,.2)'; });
  uploadArea.addEventListener('drop', function(e){
    e.preventDefault();
    this.style.borderColor='rgba(255,255,255,.2)';
    fileInput.files = e.dataTransfer.files;
    showFile();
  });

  fileInput.addEventListener('change', showFile);
  fileRemove.addEventListener('click', removeFile);

  function showFile(){
    if(fileInput.files && fileInput.files[0]){
      var file = fileInput.files[0];
      if(file.size > 10*1024*1024){
        alert('File is too large. Maximum size is 10MB.');
        fileInput.value = '';
        return;
      }
      fileNameText.textContent = '\uD83D\uDCCE ' + file.name + ' (' + (file.size/1024/1024).toFixed(1) + ' MB)';
      fileNameDiv.style.display = 'block';
    }
  }

  function removeFile(){
    fileInput.value = '';
    fileNameDiv.style.display = 'none';
  }

  document.getElementById('featured-form').addEventListener('submit', function(e){
    e.preventDefault();
    var btn = document.getElementById('submit-btn');
    var errorDiv = document.getElementById('form-error');
    var errorText = document.getElementById('error-text');
    var successDiv = document.getElementById('form-success');
    var form = this;

    btn.disabled = true;
    btn.textContent = 'Submitting...';
    btn.style.opacity = '0.6';
    errorDiv.style.display = 'none';

    var formData = new FormData(form);

    fetch('/get-featured/submit-featured.php', {
      method: 'POST',
      body: formData
    })
    .then(function(res){ return res.json(); })
    .then(function(data){
      if(data.success){
        form.style.display = 'none';
        successDiv.style.display = 'block';
      } else {
        errorText.textContent = data.message;
        errorDiv.style.display = 'block';
        btn.disabled = false;
        btn.textContent = 'Submit Application \u2014 Free';
        btn.style.opacity = '1';
      }
    })
    .catch(function(){
      errorText.textContent = 'Network error. Please try again or email us directly.';
      errorDiv.style.display = 'block';
      btn.disabled = false;
      btn.textContent = 'Submit Application \u2014 Free';
      btn.style.opacity = '1';
    });
  });
})();
</script>
